JavaScript 基础知识练习

一、变量的定义

// var - 函数作用域，可重复声明
var name = "张三";
// let - 块级作用域，不可重复声明
let age = 20;
// const - 常量，不可重新赋值
const PI = 3.14159;

二、变量类型

// 基本数据类型
let str = "字符串"; // string
let num = 42; // number
let flag = true; // boolean
let empty = null; // null
let notDefined; // undefined
let sym = Symbol("id"); // symbol
let big = 9007199254740991n; // bigint

// 引用数据类型
let arr = [1, 2, 3]; // object (数组)
let obj = {name: "李四"}; // object (对象)

三、表达式

// 算术表达式
let sum = 10 + 5 * 2; // 20
let mod = 17 % 5; // 2

// 比较表达式
let isEqual = (5 == "5"); // true (松散相等)
let isStrictEqual = (5 === "5"); // false (严格相等)

// 逻辑表达式
let logic = (true && false) || true; // true

// 三元表达式
let grade = score >= 60 ? "及格" : "不及格";

四、流程控制

// if-else 条件判断
if (score >= 90) {
  level = "优秀";
} else if (score >= 60) {
  level = "及格";
} else {
  level = "不及格";
}

// switch 多分支
switch(day) {
  case 1: result = "星期一"; break;
  case 2: result = "星期二"; break;
  default: result = "其他"; break;
}

// for 循环
for (let i = 0; i < 5; i++) { ... }

// while 循环
while (count > 0) { ... }

// for...of 遍历数组
for (let item of array) { ... }

五、函数

// 函数声明
function greet(name) {
  return "你好, " + name + "!";
}

// 函数表达式
const add = function(a, b) {
  return a + b;
};

// 箭头函数
const multiply = (a, b) => a * b;

// 默认参数
function sayHello(name = "访客") {
  return `欢迎, ${name}!`;
}

// 回调函数
function processData(arr, callback) {
  return arr.map(callback);
}
